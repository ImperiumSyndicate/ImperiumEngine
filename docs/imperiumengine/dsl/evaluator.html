298def safe_eval_expr(expr: str, context: dict[str, Any]) -> Any:
299    """
300    Avalia de forma segura uma expressão em uma DSL utilizando uma AST restrita.
301
302    Esta função substitui operadores especiais por suas equivalentes em palavras (por exemplo,
303    "→" por "implies") e cria um contexto local que inclui funções permitidas para operações
304    lógicas. Em seguida, a expressão é analisada e avaliada utilizando a classe `SafeEvaluator`.
305
306    Parameters
307    ----------
308    expr : str
309        A expressão a ser avaliada. Pode conter operadores especiais que serão convertidos.
310    context : dict[str, Any]
311        Dicionário com variáveis e funções disponíveis para a avaliação da expressão.
312
313    Returns
314    -------
315    Any
316        Resultado da avaliação da expressão.
317
318    Raises
319    ------
320    DSLError
321        Se ocorrer algum erro durante a análise ou avaliação segura da expressão.
322
323    Examples
324    --------
325    >>> contexto = {"x": 10, "y": 5}
326    >>> # A expressão utiliza operadores aritméticos e lógicos seguros.
327    >>> safe_eval_expr("x + y", contexto)
328    15
329    >>> # Exemplo utilizando um operador especial:
330    >>> safe_eval_expr("x > y and implies(x == 10, y == 5)", contexto)
331    True
332    """
333    expr = expr.replace("→", " implies ")
334    expr = expr.replace("↔", " iff ")
335    expr = expr.replace("⊕", " xor ")
336    expr = expr.replace("↑", " nand ")
337    expr = expr.replace("↓", " nor ")
338    local_context = dict(context)
339    local_context["implies"] = lambda x, y: (not x) or y
340    local_context["iff"] = lambda x, y: (x and y) or ((not x) and (not y))
341    local_context["xor"] = lambda x, y: (x and (not y)) or ((not x) and y)
342    local_context["nand"] = lambda x, y: not (x and y)
343    local_context["nor"] = lambda x, y: not (x or y)
344    try:
345        node = ast.parse(expr, mode="eval")
346        evaluator = SafeEvaluator(local_context)
347        return evaluator.visit(node.body)
348    except Exception as e:
349        raise DSLError(f"Error in safe evaluation of expression '{expr}': {e}") from e

352def safe_exec_statement(statement: str, context: dict[str, Any]) -> None:
353    """
354    Executa de forma segura uma única instrução definida por uma string.
355
356    Esta função permite apenas instruções simples, como atribuições simples a uma variável ou
357    chamadas de função restritas (por exemplo, a função `print`). Instruções com mais de um
358    comando ou estruturas não permitidas geram uma exceção `DSLError`.
359
360    Parameters
361    ----------
362    statement : str
363        A instrução a ser executada. Exemplos: "x = 5" ou "print(x)".
364    context : dict[str, Any]
365        Dicionário contendo as variáveis (e funções) que poderão ser modificadas ou utilizadas
366        durante a execução da instrução.
367
368    Returns
369    -------
370    None
371
372    Raises
373    ------
374    DSLError
375        Se a instrução não estiver no formato permitido, se houver mais de uma instrução, ou se
376        a instrução contiver elementos não permitidos.
377
378    Examples
379    --------
380    >>> contexto = {"x": 10}
381    >>> safe_exec_statement("x = x + 5", contexto)
382    >>> contexto["x"]
383    15
384    >>> # Apenas chamadas à função print são permitidas:
385    >>> safe_exec_statement("print(x)", contexto)
386    15
387    """
388    try:
389        tree = ast.parse(statement, mode="exec")
390    except Exception as e:
391        raise DSLError(f"Error parsing statement '{statement}': {e}") from e
392    if len(tree.body) != 1:
393        raise DSLError("Only one statement is permitted.")
394    node = tree.body[0]
395    evaluator = SafeEvaluator(context)
396    if isinstance(node, ast.Assign):
397        if len(node.targets) != 1 or not isinstance(node.targets[0], ast.Name):
398            raise DSLError("Assignment target must be a single variable.")
399        var_name = node.targets[0].id
400        value = evaluator.visit(node.value)
401        context[var_name] = value
402    elif isinstance(node, ast.Expr):
403        expr = node.value
404        if isinstance(expr, ast.Call):
405            if isinstance(expr.func, ast.Name) and expr.func.id == "print":
406                args = [evaluator.visit(arg) for arg in expr.args]
407                kwargs = {kw.arg: evaluator.visit(kw.value) for kw in expr.keywords if kw.arg is not None}
408                print(*args, **kwargs)
409            else:
410                raise DSLError("Only 'print' function calls are permitted.")
411        else:
412            raise DSLError("Only function calls are permitted in expressions.")
413    else:
414        raise DSLError("Only simple assignments or permitted function calls are allowed.")

Avalia expressões de uma DSL de forma segura utilizando uma AST restrita.

Esta classe percorre a árvore sintática abstrata (AST) gerada a partir de uma expressão e avalia os nós permitidos, garantindo que somente operações seguras sejam executadas. Operações não permitidas ou nós inesperados geram uma exceção DSLError.

A classe permite operações aritméticas simples, operações unárias, operações booleanas, comparações, além de chamadas a funções permitidas. As funções autorizadas estão definidas na variável de classe ALLOWED_FUNCTIONS.

  • ALLOWED_FUNCTIONS (ClassVar[set[str]]): Conjunto de nomes de funções que são permitidas nas expressões. Valores permitidos: {"implies", "iff", "xor", "nand", "nor"}.
  • context (dict[str, Any]): Dicionário que representa o contexto de avaliação, contendo variáveis e funções disponíveis.
Parameters
  • context (dict[str, Any]): Dicionário contendo as variáveis e funções que poderão ser utilizadas durante a avaliação.
Examples

>>> import ast

300    """
301    Avalia de forma segura uma expressão em uma DSL utilizando uma AST restrita.
302
303    Esta função substitui operadores especiais por suas equivalentes em palavras (por exemplo,
304    "→" por "implies") e cria um contexto local que inclui funções permitidas para operações
305    lógicas. Em seguida, a expressão é analisada e avaliada utilizando a classe `SafeEvaluator`.
306
307    Parameters
308    ----------
309    expr : str
310        A expressão a ser avaliada. Pode conter operadores especiais que serão convertidos.
311    context : dict[str, Any]
312        Dicionário com variáveis e funções disponíveis para a avaliação da expressão.
313
314    Returns
315    -------
316    Any
317        Resultado da avaliação da expressão.
318
319    Raises
320    ------
321    DSLError
322        Se ocorrer algum erro durante a análise ou avaliação segura da expressão.
323
324    Examples
325    --------
326    >>> contexto = {"x": 10, "y": 5}
327    >>> # A expressão utiliza operadores aritméticos e lógicos seguros.
328    >>> safe_eval_expr("x + y", contexto)
329    15
330    >>> # Exemplo utilizando um operador especial:
331    >>> safe_eval_expr("x > y and implies(x == 10, y == 5)", contexto)
332    True
333    """
334    expr = expr.replace("→", " implies ")
335    expr = expr.replace("↔", " iff ")
336    expr = expr.replace("⊕", " xor ")
337    expr = expr.replace("↑", " nand ")
338    expr = expr.replace("↓", " nor ")
339    local_context = dict(context)
340    local_context["implies"] = lambda x, y: (not x) or y
341    local_context["iff"] = lambda x, y: (x and y) or ((not x) and (not y))
342    local_context["xor"] = lambda x, y: (x and (not y)) or ((not x) and y)
343    local_context["nand"] = lambda x, y: not (x and y)
344    local_context["nor"] = lambda x, y: not (x or y)
345    try:
346        node = ast.parse(expr, mode="eval")
347        evaluator = SafeEvaluator(local_context)
348        return evaluator.visit(node.body)
349    except Exception as e:
350        raise DSLError(f"Error in safe evaluation of expression '{expr}': {e}") from e

354    """
355    Executa de forma segura uma única instrução definida por uma string.
356
357    Esta função permite apenas instruções simples, como atribuições simples a uma variável ou
358    chamadas de função restritas (por exemplo, a função `print`). Instruções com mais de um
359    comando ou estruturas não permitidas geram uma exceção `DSLError`.
360
361    Parameters
362    ----------
363    statement : str
364        A instrução a ser executada. Exemplos: "x = 5" ou "print(x)".
365    context : dict[str, Any]
366        Dicionário contendo as variáveis (e funções) que poderão ser modificadas ou utilizadas
367        durante a execução da instrução.
368
369    Returns
370    -------
371    None
372
373    Raises
374    ------
375    DSLError
376        Se a instrução não estiver no formato permitido, se houver mais de uma instrução, ou se
377        a instrução contiver elementos não permitidos.
378
379    Examples
380    --------
381    >>> contexto = {"x": 10}
382    >>> safe_exec_statement("x = x + 5", contexto)
383    >>> contexto["x"]
384    15
385    >>> # Apenas chamadas à função print são permitidas:
386    >>> safe_exec_statement("print(x)", contexto)
387    15
388    """
389    try:
390        tree = ast.parse(statement, mode="exec")
391    except Exception as e:
392        raise DSLError(f"Error parsing statement '{statement}': {e}") from e
393    if len(tree.body) != 1:
394        raise DSLError("Only one statement is permitted.")
395    node = tree.body[0]
396    evaluator = SafeEvaluator(context)
397    if isinstance(node, ast.Assign):
398        if len(node.targets) != 1 or not isinstance(node.targets[0], ast.Name):
399            raise DSLError("Assignment target must be a single variable.")
400        var_name = node.targets[0].id
401        value = evaluator.visit(node.value)
402        context[var_name] = value
403    elif isinstance(node, ast.Expr):
404        expr = node.value
405        if isinstance(expr, ast.Call):
406            if isinstance(expr.func, ast.Name) and expr.func.id == "print":
407                args = [evaluator.visit(arg) for arg in expr.args]
408                kwargs = {kw.arg: evaluator.visit(kw.value) for kw in expr.keywords if kw.arg is not None}
409                print(*args, **kwargs)
410            else:
411                raise DSLError("Only 'print' function calls are permitted.")
412        else:
413            raise DSLError("Only function calls are permitted in expressions.")
414    else:
415        raise DSLError("Only simple assignments or permitted function calls are allowed.")
